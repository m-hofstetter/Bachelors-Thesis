\begin{algorithm}[H]
$r \leftarrow 0$\\
$w,b \leftarrow$ length of $A-1$ \\
\vspace{1.5mm}
\While{ $r\leq w$}{
    \vspace{0.7mm}
    \vspace{0.7mm}
    \If{$A[w]\leq -1$}{
        \vspace{0.7mm}
        switch values of $A[w]$ and $A[r]$\\
        $r\leftarrow r+1$ 
    }
    \vspace{1mm}
    \If{$A[w]=0$ }{
        \vspace{0.7mm}
        $w\leftarrow w-1$
    }
    \vspace{1mm}
    \If{$A[w]\geq1$}{
        \vspace{0.7mm}
        switch values of $A[w]$ and $A[b]$\\
        $w\leftarrow w-1$ \\
        $b\leftarrow b-1$ 
    }
}
\end{algorithm}
